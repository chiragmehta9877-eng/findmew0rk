import { MetadataRoute } from 'next'

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: '*',
      allow: '/',
      disallow: ['/api/', '/dashboard/', '/private/'], // Private pages ko hide karein
    },
    sitemap: 'https://findmew0rk.com/sitemap.xml',
  }
}
